# ===================== 新增的关键配置（仅改这里） =====================
# 第二个服务（子域名 music.luyunxiao.xyz → 3000端口）
server {
    listen 443 ssl;
    server_name music.luyunxiao.xyz;  # 可替换为你想要的子域名（如 app.luyunxiao.xyz）
    ssl_certificate /etc/nginx/ssl/luyunxiao.xyz.pem;  # 复用原有证书（通配符证书直接用，单域名需重新申请）
    ssl_certificate_key /etc/nginx/ssl/luyunxiao.xyz.key;  # 复用原有证书
    
    ssl_protocols TLSv1.2 TLSv1.3;
    ssl_ciphers ECDHE-ECDSA-AES128-GCM-SHA256:ECDHE-RSA-AES128-GCM-SHA256:ECDHE-ECDSA-AES256-GCM-SHA384:ECDHE-RSA-AES256-GCM-SHA384;
    ssl_prefer_server_ciphers on;

    location / {
        proxy_pass http://127.0.0.1:3000;  # 固定改为你的新服务端口3000
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
    }
}

# 第二个服务的HTTP自动跳转HTTPS
server {
    listen 80;
    server_name music.luyunxiao.xyz;  # 与上面子域名保持一致
    return 301 https://$host$request_uri;
}


server {
    listen 80;
    server_name luyunxiao.xyz www.luyunxiao.xyz;
    return 301 https://$host$request_uri;
}

# HTTPS配置（反向代理Halo）
server {
    listen 443 ssl;
    server_name luyunxiao.xyz www.luyunxiao.xyz;
    
    # 新增这两行，设置上传文件的大小限制
    client_max_body_size 100m;  # 允许的最大上传文件大小（m=MB，也可以用g=GB）
    client_body_buffer_size 128k;  # 上传文件的临时缓冲区大小（辅助参数）
    
    # SSL证书路径（对应./ssl目录下的文件）
    ssl_certificate /etc/nginx/ssl/luyunxiao.xyz.pem;
    ssl_certificate_key /etc/nginx/ssl/luyunxiao.xyz.key;

    # SSL优化配置
    ssl_protocols TLSv1.2 TLSv1.3;
    ssl_ciphers ECDHE-ECDSA-AES128-GCM-SHA256:ECDHE-RSA-AES128-GCM-SHA256:ECDHE-ECDSA-AES256-GCM-SHA384:ECDHE-RSA-AES256-GCM-SHA384;
    ssl_prefer_server_ciphers on;

    # 反向代理到Halo
    location / {
        proxy_pass http://halo:8090;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
    }
}
